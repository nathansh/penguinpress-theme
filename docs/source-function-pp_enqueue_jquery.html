 10 /**
 11  * Enqueue jQuery from Google CDN with fallback to local WordPress
 12  * First we have a really complex method of loading jQuery from the CDN or local
 13  * should the CDN fail. The code is adapted from https://gist.github.com/wpsmith/4083811.
 14  * Then we load basic styles.
 15  *
 16  * @package pp
 17  * @subpackage boilerplate-theme_filters+hooks
 18  * @internal only called by `wp_enqueue_scripts` action
 19  *
 20  * @link https://gist.github.com/wpsmith/4083811.
 21  * @link http://codex.wordpress.org/Function_Reference/wp_enqueue_script
 22  * @link http://codex.wordpress.org/Function_Reference/wp_register_script
 23  * @link http://codex.wordpress.org/Function_Reference/wp_deregister_script
 24  * @link http://codex.wordpress.org/Function_Reference/get_bloginfo
 25  * @link http://codex.wordpress.org/Function_Reference/is_wp_error
 26  * @link http://codex.wordpress.org/Function_Reference/set_transient
 27  * @link http://codex.wordpress.org/Function_Reference/get_transient
 28  *
 29  * @uses get_transient()        Get the value of a transient.
 30  * @uses set_transient()        Set/update the value of a transient.
 31  * @uses is_wp_error()          Check whether the passed variable is a WordPress Error.
 32  * @uses get_bloginfo()         returns information about your site.
 33  * @uses wp_deregister_script() Deregisters javascripts for use with wp_enqueue_script() later.
 34  * @uses wp_register_script()   Registers javascripts for use with wp_enqueue_script() later.
 35  * @uses wp_enqueue_script()    Enqueues javascript.
 36  */
 37 function pp_enqueue_jquery() {
 38 
 39     $script_location = 'js/main.js';
 40 
 41     // Setup Google URI, default
 42     $protocol = ( isset( $_SERVER['HTTPS'] ) && 'on' == $_SERVER['HTTPS'] ) ? 'https' : 'http';
 43     // Get Latest Version
 44     $url = $protocol . '://code.jquery.com/jquery-1.11.1.min.js';
 45 
 46     // Get Specific Version
 47     //$url      = $protocol . '://ajax.googleapis.com/ajax/libs/jquery/1.8.2/jquery.min.js';
 48 
 49     // Setup WordPress URI
 50     $wpurl =  get_bloginfo('template_directory') . '/js/jquery-1.11.1.min.js';
 51 
 52     // Setup version
 53     $ver = null;
 54 
 55     // Deregister WordPress default jQuery
 56     wp_deregister_script( 'jquery' );
 57 
 58     // Check transient, if false, set URI to WordPress URI
 59     delete_transient( 'google_jquery' );
 60 
 61     if ( 'false' == ( $google = get_transient( 'google_jquery' ) ) ) {
 62         $url = $wpurl;
 63     }
 64     // Transient failed
 65     elseif ( false === $google ) {
 66         // Ping Google
 67         $resp = wp_remote_head( $url );
 68 
 69         // Use Google jQuery
 70         if ( ! is_wp_error( $resp ) && 200 == $resp['response']['code'] ) {
 71             // Set transient for 5 minutes
 72             set_transient( 'google_jquery', 'true', 60 * 5 );
 73         }
 74 
 75         // Use WordPress jQuery
 76         else {
 77             // Set transient for 5 minutes
 78             set_transient( 'google_jquery', 'false', 60 * 5 );
 79 
 80             // Use WordPress URI
 81             $url = $wpurl;
 82 
 83             // Set jQuery Version, WP stanards
 84             $ver = '1.8.2';
 85         }
 86     }
 87 
 88     // Register surefire jQuery
 89     wp_register_script( 'jquery', $url, array(), $ver, true );
 90 
 91     // Enqueue jQuery
 92     wp_enqueue_script( 'jquery' );
 93 
 94     // Now load basic site js
 95     wp_enqueue_script('basic', get_bloginfo('template_directory') . '/' . $script_location , array('jquery'), '1.0');
 96 
 97     // Add some site information to a WP js object
 98     $wp_object = array(
 99         'templateUrl' => get_bloginfo('template_url'),
100         'stylesheetUrl' => get_bloginfo('stylesheet_url'),
101         'stylesheetDirectory' => get_bloginfo('stylesheet_directory'),
102         'siteName' => get_bloginfo('name'),
103         'description' => get_bloginfo('description'),
104         'currentTheme' => wp_get_theme(),
105         'url' => get_bloginfo('url'),
106     );
107     wp_localize_script('basic', 'ppWP', $wp_object);
108 
109 
110     // For comment reply form
111     if ( is_singular() && get_option( 'thread_comments' ) && comments_open() ) {
112         wp_enqueue_script( 'comment-reply' );
113     }
114 
115 }
